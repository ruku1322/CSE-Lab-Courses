
#include <stdio.h>
#include <string.h>

int main() {
    char word[20];
    char b = 'z';
    printf("Enter a string: ");
    fgets(word, sizeof(word), stdin);

    int length = strlen(word);


    for (int i = 0; i < length; i++) {
        if (word[i] == '%') {

            for (int j = i + 1; j < length; j++) {
                if (word[j] == '&') {
                    b = 'i';
                    break;
                }
            }
            break;
        }
    }

    if (b == 'i') {
        printf("input\n");
    } else {
        printf("not an input\n");
    }

    return 0;
}
